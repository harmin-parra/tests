server {
    listen 80;
    listen [::]:80;

    server_name _;

    root /www/;
    index index.html;

    # Directory listing ONLY
    location / {
        index index.html;
        autoindex on;
        autoindex_format html;
        autoindex_exact_size off;   # shows sizes in KB/MB
        autoindex_localtime on;     # local time instead of UTC


        gzip off;
        sub_filter_types text/html;
        sub_filter '<h1>Index of ' '<h1>Index of <style>
            body {
                font-size: 20px;
            }
            a {
                font-size: 22px;
                font-weight: 500;
            }
        </style>';
        sub_filter_once on;
    }
}
